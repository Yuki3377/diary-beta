if (request.method !== "POST") { return new Response("APIは正常稼働中！アプリからリクエストしてね。", { headers }); } try { // Cloudflareの環境変数に GEMINI_API_KEY が設定されていることを確認してください const API_KEY = (env.GEMINI_API_KEY || "").trim(); const requestData = await request.json(); // --- プロンプト（命令文）の組み立て --- let systemInstruction = ""; let userMessage = ""; // 受け取ったデータ（共通） const ctx = requestData.context || {}; const name = ctx.name || "私"; const character = ctx.character || "お姉さん"; const customerCall = ctx.customer || "お客様"; const traits = (ctx.traits || []).join("、"); // ★ 1. 日記モード if (requestData.promptType === "index") { const mood = ctx.mood || "普通";
